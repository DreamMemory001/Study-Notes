#### 在n个球中，任意取出m个球(不放回)，共有多少种取法。
```java
package LanQiao;

public class Main {

    public static int f(int n, int m) {

        if (n < m){
            return 0;
        }
        if (n == m){
            return 1;
        }
        if (m == 0) {
            return 1;
        }
            return f(n - 1, m - 1) + f(n - 1, m);////设置一个特殊球，进行划分，第一种包含，第二种不包含。
    }

    public static void main(String[] args) {
       int k = f(3,2);
       System.out.println(k);
    }
}
```
#### 求n个元素的全排列
```java
package LanQiao;

//求n个元素的全排列

public class Two {

    public static void f(char[] data,int k){

        //k：当前的交换位置，与其后的元素交换
        if (k == data.length) {
            for (int i = 0; i < data.length; i++) {
                System.out.print(data[i] + " ");                  //打印
            }
            System.out.println();
        }

        for(int i = k;i < data.length;i++) {
            {                                   //试探，交换位置
                char t = data[k];                 
                data[k] = data[i];
                data[i] = t;
            }

            f(data, k + 1);                   //递归

            {                                    //回溯
                char t = data[k];
                data[k] = data[i];
                data[i] = t;
            }
        }
    }

    public static void main(String[] args) {
        char[] data = "ABC".toCharArray();
        f(data,0);
    }
}
```
#### 求两个串的最大公共子序列的长度
```java
package LanQiao;

//求两个串的最大公共子序列的长度

public class Three {

    public static int f(String s1,String s2){

        if(s1.length() == 0 || s2.length() == 0){             //两个串的长度是0
            return 0;
        }

        if (s1.charAt(0) == s2.charAt(0)){                    //两个串的首元素相等，将剩下的元素进行递归
            return f(s1.substring(1),s2.substring(1)) + 1;
        }

        else return Math.max(f(s1.substring(1),s2),f(s1,s2.substring(1))); //两个串的首元素不同，将s1与s2的剩下元素进行比较或者是将s1的剩下元素与s2进行比较。
    }
    public static void main(String[] args){
        int k = f("abc","abecd");
        System.out.println(k);
    }
}
```
